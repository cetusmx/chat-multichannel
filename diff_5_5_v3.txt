+ *         schema:
+ *           type: string
+ *           format: date-time
+ *         required: true
+ *     responses:
+ *       200:
+ *         description: Productivity metrics retrieved successfully.
+ *       400:
+ *         description: Validation error.
+ *       401:
+ *         description: Unauthorized.
+ *       403:
+ *         description: Forbidden.
+ */
+router.get('/productivity', authorize('ADMIN', 'COORDINATOR'), async (req, res, next) => {
+  try {
+    const { startDate, endDate } = req.query;
+    if (!startDate || !endDate) {
+      throw new ApiError(400, 'startDate and endDate are required');
+    }
+
+    const start = new Date(startDate);
+    const end = new Date(endDate);
+
+    if (isNaN(start.getTime()) || isNaN(end.getTime())) {
+      throw new ApiError(400, 'Invalid startDate or endDate format');
+    }
+
+    const diffMs = end.getTime() - start.getTime();
+    if (diffMs < 0) {
+      throw new ApiError(400, 'endDate must be after startDate');
+    }
+
+    // 1 year in ms (approx)
+    const MAX_DATE_RANGE_MS = 366 * 24 * 60 * 60 * 1000;
+    if (diffMs > MAX_DATE_RANGE_MS) {
+      throw new ApiError(400, 'Date range cannot exceed 1 year');
+    }
+
+    const metrics = await metricsService.getVendorProductivityMetrics(req.user.tenantId, startDate, endDate);
+    res.json({ data: metrics });
+  } catch (error) {
+    next(error);
+  }
+});

+const { PrismaClient } = require('@prisma/client');
+const prisma = new PrismaClient();
+
+/**
+ * Get vendor productivity metrics
+ * @param {string} tenantId 
+ * @param {Date|string} startDate 
+ * @param {Date|string} endDate 
+ * @returns {Promise<Array>}
+ */
+const getVendorProductivityMetrics = async (tenantId, startDate, endDate) => {
+  if (!tenantId) {
+    throw new Error('tenantId is required');
+  }
+
+  const start = new Date(startDate);
+  const end = new Date(endDate);
+
+  // 1. One efficient raw SQL query to get all metrics
+  const metricsData = await prisma.$queryRaw`
+    WITH client_first AS (
+      SELECT c.id as conversation_id, MIN(m.created_at) as first_msg_time
+      FROM conversations c
+      JOIN messages m ON c.id = m.conversation_id
+      WHERE c.tenant_id = ${tenantId}
+        AND c.created_at >= ${start}
+        AND c.created_at <= ${end}
+        AND m.sender_type = 'CLIENT'
+      GROUP BY c.id
+    ),
+    vendor_first_reply AS (
+      SELECT m.conversation_id, m.sender_id as vendor_id, MIN(m.created_at) as first_reply_time
+      FROM client_first cf
+      JOIN messages m ON cf.conversation_id = m.conversation_id
+      WHERE m.sender_type = 'VENDOR'
+        AND m.created_at > cf.first_msg_time
+        AND m.sender_id IS NOT NULL
+      GROUP BY m.conversation_id, m.sender_id
+    ),
+    response_times AS (
+      SELECT 
+        vfr.vendor_id,
+        AVG(EXTRACT(EPOCH FROM (vfr.first_reply_time - cf.first_msg_time))) as avg_response_time
+      FROM client_first cf
+      JOIN vendor_first_reply vfr ON cf.conversation_id = vfr.conversation_id
+      GROUP BY vfr.vendor_id
+    ),
+    vendor_participation AS (
+      SELECT DISTINCT m.conversation_id, m.sender_id as vendor_id
+      FROM conversations c
+      JOIN messages m ON c.id = m.conversation_id
+      WHERE c.tenant_id = ${tenantId}
+        AND c.created_at >= ${start}
+        AND c.created_at <= ${end}
+        AND m.sender_type = 'VENDOR'
+        AND m.sender_id IS NOT NULL
+    ),
+    total_handled AS (
+      SELECT vendor_id, COUNT(conversation_id)::int as total_chats
+      FROM vendor_participation
+      GROUP BY vendor_id
+    ),
+    closed_handled AS (
+      SELECT vp.vendor_id, COUNT(vp.conversation_id)::int as closed_chats
+      FROM vendor_participation vp
+      JOIN conversations c ON vp.conversation_id = c.id
+      WHERE c.status = 'CLOSED'
+      GROUP BY vp.vendor_id
+    )
+    SELECT 
+      th.vendor_id as "vendorId",
+      th.total_chats as "totalChatsHandled",
+      COALESCE(ch.closed_chats, 0) as "closedChats",
+      rt.avg_response_time as "avgResponseTime"
+    FROM total_handled th
+    LEFT JOIN closed_handled ch ON th.vendor_id = ch.vendor_id
+    LEFT JOIN response_times rt ON th.vendor_id = rt.vendor_id
+  `;
+
+  // 2. Fetch vendor details for the ones that participated
+  const vendorIds = metricsData.map(m => m.vendorId);
+  
+  const vendors = await prisma.user.findMany({
+    where: { 
+      id: { in: vendorIds },
+      tenantId 
+    },
+    select: { id: true, name: true, email: true }
+  });
+
+  const vendorDict = {};
+  vendors.forEach(v => {
+    vendorDict[v.id] = v;
+  });
+
+  // 3. Format the result
+  return metricsData.map(item => {
+    const vendor = vendorDict[item.vendorId] || { name: 'Unknown', email: '' };
+    const totalChats = Number(item.totalChatsHandled || 0);
+    const closedChats = Number(item.closedChats || 0);
+    const avgResponseTime = item.avgResponseTime;
+    
+    return {
+      vendorId: item.vendorId,
+      name: vendor.name,
+      email: vendor.email,
+      totalChatsHandled: totalChats,
+      resolutionRate: totalChats > 0 ? (closedChats / totalChats) : 0,
+      averageResponseTime: (avgResponseTime !== null && avgResponseTime !== undefined) ? Number(avgResponseTime) : null
+    };
+  });
+};
+
+module.exports = {
+  getVendorProductivityMetrics
+};

++const { PrismaClient } = require('@prisma/client');
++const prisma = new PrismaClient();
++
++/**
++ * Get vendor productivity metrics
++ * @param {string} tenantId 
++ * @param {Date|string} startDate 
++ * @param {Date|string} endDate 
++ * @returns {Promise<Array>}
++ */
++const getVendorProductivityMetrics = async (tenantId, startDate, endDate) => {
++  if (!tenantId) {
++    throw new Error('tenantId is required');
++  }
++
++  const start = new Date(startDate);
++  const end = new Date(endDate);
++
++  // 1. Total assigned chats per vendor
++  const totalChats = await prisma.conversation.groupBy({
++    by: ['vendorId'],
++    _count: { id: true },
++    where: {
++      tenantId,
++      vendorId: { not: null },
++      createdAt: { gte: start, lte: end }
++    }
++  });
++
++  // 2. Closed chats per vendor
++  const closedChats = await prisma.conversation.groupBy({
++    by: ['vendorId'],
++    _count: { id: true },
++    where: {
++      tenantId,
++      vendorId: { not: null },
++      status: 'CLOSED',
++      createdAt: { gte: start, lte: end }
++    }
++  });
++
++  // 3. Average response time per vendor (raw SQL)
++  const responseTimes = await prisma.$queryRaw`
++    WITH client_first AS (
++      SELECT c.id as conversation_id, c.vendor_id, MIN(m.created_at) as first_msg_time
++      FROM conversations c
++      JOIN messages m ON c.id = m.conversation_id
++      WHERE c.tenant_id = ${tenantId}
++        AND c.created_at >= ${start}
++        AND c.created_at <= ${end}
++        AND c.vendor_id IS NOT NULL
++        AND m.sender_type = 'CLIENT'
++      GROUP BY c.id, c.vendor_id
++    ),
++    vendor_first_reply AS (
++      SELECT m.conversation_id, MIN(m.created_at) as first_reply_time
++      FROM client_first cf
++      JOIN messages m ON cf.conversation_id = m.conversation_id
++      WHERE m.sender_type = 'VENDOR'
++        AND m.created_at > cf.first_msg_time
++      GROUP BY m.conversation_id
++    )
++    SELECT 
++      cf.vendor_id as "vendorId",
++      AVG(EXTRACT(EPOCH FROM (vfr.first_reply_time - cf.first_msg_time))) as "avgResponseTime"
++    FROM client_first cf
++    JOIN vendor_first_reply vfr ON cf.conversation_id = vfr.conversation_id
++    GROUP BY cf.vendor_id
++  `;
++
++  // Get all vendors for this tenant to avoid omitting underperforming ones
++  const allVendors = await prisma.user.findMany({
++    where: {
++      tenantId,
++      role: 'VENDOR'
++    },
++    select: { id: true }
++  });
++
++  // Merge the results
++  const vendorMap = {};
++
++  // Initialize map with all vendors
++  allVendors.forEach(vendor => {
++    vendorMap[vendor.id] = {
++      vendorId: vendor.id,
++      totalChatsHandled: 0,
++      resolutionRate: 0,
++      averageResponseTime: null
++    };
++  });
++
++  totalChats.forEach(item => {
++    if (vendorMap[item.vendorId]) {
++      vendorMap[item.vendorId].totalChatsHandled = item._count.id;
++    }
++  });
++
++  closedChats.forEach(item => {
++    if (vendorMap[item.vendorId]) {
++      const total = vendorMap[item.vendorId].totalChatsHandled;
++      vendorMap[item.vendorId].resolutionRate = total > 0 ? (item._count.id / total) : 0;
++    }
++  });
++
++  responseTimes.forEach(item => {
++    if (vendorMap[item.vendorId] && item.avgResponseTime != null) {
++      // Postgres returns AVG as a string or BigInt/Number depending on the driver, ensure it's a number
++      vendorMap[item.vendorId].averageResponseTime = Number(item.avgResponseTime);
++    }
++  });
++
++  return Object.values(vendorMap);
++};
